After the CD ejects, fix the initrd, install grub in the right place, fix menu.lst, config networking

Alt+F2

# fix initrd
echo megaraid_sas >> /etc/mkinitramfs/modules
cd /boot/
cp initrd.img-2.6.15-26-amd64-server initrd.img-2.6.15-26-amd64-server.old
mkinitramfs -o /boot/initrd.img-2.6.15-26-amd64-server 2.6.15-26-amd64-server

# install grub
grub-install /dev/sdc

# change (hd4,0) to (hd0,0) in menu.lst
cp /boot/grub/menu.lst /boot/grub/menu.lst.orig
sed -e 's/hd4/hd0/g' /boot/grub/menu.lst.orig > /boot/grub/menu.lst

# edit the /boot/grub/device.map
cd /boot/grub/
cp device.map device.map.orig
rm device.map
vi device.map
# add the following line and save and close the file (:wq)
(hd0) /dev/sde

# configure networking
echo '
# The Ethernet network interface
auto eth0
iface eth0 inet dhcp

auto eth1
iface eth1 inet dhcp
' >> /etc/network/interfaces

touch /etc/resolv.conf

# exit the chroot
exit

# unmount /target
umount /target

# reboot (optional)
Alt+SysRq+B
