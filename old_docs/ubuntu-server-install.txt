# setup locales
$ sudo locale-gen en_US
$ sudo locale-gen en_US.UTF-8

# installed the lamp stack
$ sudo apt-get install mysql-server mysql-admin apache2 php5 libapache2-mod-php5 libapache2-mod-auth-mysql php5-mysql

# set the root password for mysql
$ sudo mysqladmin -u root password your-new-password
$ sudo /etc/init.d/mysql restart

# restart apache
$ sudo /etc/init.d/apache2 restart

# changed the document root for the default website
$ sudo vi /etc/apache2/sites-available/default
# changed
DocumentRoot /var/www/
# to
DocumentRoot /var/www/apache2-default
$ sudo /etc/init.d/apache2 restart

# Added ServerName above SeverRoot in /etc/apache2/apache2.conf
