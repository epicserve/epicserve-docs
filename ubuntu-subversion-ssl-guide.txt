# Used information from, https://help.ubuntu.com/community/Subversion

# Installed libapache2-svn

apt-get install libapache2-svn

# edited the virtual host file and added the following

<Location /code/ciy>
	DAV svn
	SVNPath /var/www/code/ciy
	AuthType Basic
	AuthName "CIY Subversion Repository"
	AuthUserFile /etc/subversion/passwd
	Require valid-user
</Location>

# changed the owner of the files of the actual repository
chown -R www-data:www-data
